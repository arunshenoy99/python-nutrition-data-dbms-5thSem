CREATE TABLE NUTRITION(FID INT PRIMARY KEY,NAME VARCHAR(255),SERVING_SIZE FLOAT,CALORIES FLOAT)
CREATE TABLE FAT(FID INT REFERENCES nutrition(FID) ON DELETE CASCADE,TOTAL_FAT FLOAT,SATURATED_FAT FLOAT)
CREATE TABLE VITAMIN (FID INT REFERENCES nutrition(FID) ON DELETE CASCADE,A FLOAT,A_RAE FLOAT,B12 FLOAT,B6 FLOAT,C FLOAT,D FLOAT,E FLOAT,K FLOAT)
CREATE TABLE PROTEIN (FID INT REFERENCES NUTRITION(FID) ON DELETE CASCADE,PROTEIN FLOAT)
CREATE TABLE MINERAL (FID INT REFERENCES NUTRITION(FID),SODIUM FLOAT,CALCIUM FLOAT,IRON FLOAT,POTASSIUM FLOAT,CARB FLOAT,WATER FLOAT)
